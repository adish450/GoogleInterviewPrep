class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] preProd = new int[n];
        int[] suffProd = new int[n];

        // fill preProd
        preProd[0] = 1;
        for (int i = 1; i < n; i++) {
            preProd[i] = preProd[i-1] * nums[i-1];
        }

        // fill suffProd
        suffProd[n-1] = 1;
        for (int i = n - 2 ; i >= 0; i--) {
            suffProd[i] = suffProd[i+1] * nums[i+1];
        }

        int[] ans = new int[n];

        for (int i = 0 ; i < n; i++) {
            ans[i] = preProd[i] * suffProd[i];
        }

        return ans;

    }
}
