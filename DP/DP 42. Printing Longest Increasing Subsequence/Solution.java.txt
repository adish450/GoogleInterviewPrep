import java.util.TreeSet;

public class Solution {

	public static int longestIncreasingSubsequence(int arr[]) {
		//Your code goes here
		int n = arr.length;

		TreeSet<Integer> set = new TreeSet<>();
		set.add(arr[0]);
		for (int i = 1; i < n; i++) {
			if (arr[i] > set.last()) {
				set.add(arr[i]);
			} else {
				int c = set.ceiling(arr[i]);
				set.remove(c);
				set.add(arr[i]);
			}
		}

		return set.size();
	}
}
