import java.util.* ;
import java.io.*; 
import java.util.*;
public class Solution {
	public static int maximumNonAdjacentSum(ArrayList<Integer> nums) {
		// Write your code here.
		int n = nums.size();
		int[] dpArray = new int[n];
		Arrays.fill(dpArray, -1);
		return maximumNonAdjacentSum(n-1,nums,dpArray);
	}

	private static int maximumNonAdjacentSum(int index, ArrayList<Integer> nums, int[] dpArray) {

		if (index == 0){
			return nums.get(index);
		}

		if (index < 0){
			return 0;
		}

		if(dpArray[index] != -1) {
			return dpArray[index];
		}

		//take
		int c1 = maximumNonAdjacentSum(index-2, nums, dpArray) + nums.get(index);

		//not take
		int c2 = maximumNonAdjacentSum(index - 1, nums, dpArray);

		return dpArray[index] = Math.max(c1,c2);

	}
}
