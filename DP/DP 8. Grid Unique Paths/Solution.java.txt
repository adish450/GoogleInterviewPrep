import java.util.* ;
import java.io.*; 
public class Solution {
	public static int uniquePaths(int m, int n) {
		// Write your code here.
		int[][] dpArray = new int[m][n];
		dpArray[m-1][n-1] = 1;
		for(int i = m-1; i >=0; i--) {
			for(int j= n-1; j >=0;j--) {
				if (i + 1 < m) {
					dpArray[i][j] += dpArray[i+1][j];
				}
				
				if (j + 1 < n) {
					dpArray[i][j] += dpArray[i][j+1];
				}
			}
		}

		return dpArray[0][0];
	}
}
